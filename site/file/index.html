Filesystem Hierarchy Standard

Search

# Finding all files containing a text string on Linux
grep -rnw '/path/to/somewhere/' -e "pattern"

Extract Files

tar

# tar
tar -xzf <file.tar.gz>

zip

# zip
zip -r my_folder.zip my_folder
unzip my_folder.zip

7z

# 7z
yum install -y p7zip
# [e]xtract files from archive
7za e <file.7z>
# [a]dd files from archive
7za a <file.7z>
